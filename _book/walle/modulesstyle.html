  • Loading

"style": {
    "overlay": {
        遮罩层
    },
    "content": {
        内容区域
    },
    "vertices": {
        节点定义

        // size可以设置Loading的大小
        "size": "30%", 

        // elements用来定义是节点数目与颜色，elements为数字时定义节点数目
        // elements为数组时，数组的长度定义节点的数目，数组子项为颜色属性时代表每个节点的颜色，用来定制彩色Loading
        "elements": ["red", "green", "blue", "yellow", "orange"] //
    }
}
